```sql
-- Tenant isolation with organization-based access
CREATE TABLE tenants (
    id UUID PRIMARY KEY,
    organization_id VARCHAR(50) UNIQUE NOT NULL,
    brand_config JSONB,
    api_credentials JSONB,
    created_at TIMESTAMP DEFAULT NOW(),
    status VARCHAR(20) DEFAULT 'active'
);

-- Client-specific users with tenant isolation
CREATE TABLE users (
    id UUID PRIMARY KEY,
    tenant_id UUID REFERENCES tenants(id),
    email VARCHAR(255),
    phone VARCHAR(20),
    kyc_status VARCHAR(20),
    created_at TIMESTAMP DEFAULT NOW()
);

-- Tenant-specific transaction records
CREATE TABLE transactions (
    id UUID PRIMARY KEY,
    tenant_id UUID REFERENCES tenants(id),
    user_id UUID REFERENCES users(id),
    transaction_type VARCHAR(20),
    amount DECIMAL(15,2),
    created_at TIMESTAMP DEFAULT NOW()
);
```

#### **API Gateway**
```javascript
// Multi-tenant API routing
app.use('/api', (req, res, next) => {
    const tenantId = req.headers['x-tenant-id'];
    const organizationId = req.headers['x-org-id'];
    
    // Tenant validation
    validateTenant(tenantId, organizationId).then(tenant => {
        req.tenant = tenant;
        next();
    }).catch(error => {
        res.status(403).json({ error: 'Invalid tenant' });
    });
});

// Tenant-specific route handlers
app.post('/api/v1/buy', async (req, res) => {
    const { tenant } = req;
    const userId = req.user.id;
    
    // Process purchase within tenant context
    const result = await processPurchase(tenant, userId, req.body);
    res.json(result);
});
```
